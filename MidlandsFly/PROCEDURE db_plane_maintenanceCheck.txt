CREATE PROCEDURE db_plane_maintenanceCheck @regnumber varchar(6)
as
SELECT CASE 
            WHEN FLYING_HOURS - LAST_MAINTENANCE >= 200
               THEN 1 
               ELSE 0 
       END as TOTFORMAIN
FROM Cargo_Aircraft
where AIRCRAFT_NUMBER=@regnumber

union
SELECT CASE 
            WHEN FLYING_HOURS - LAST_MAINTENANCE >= 200
               THEN 1 
               ELSE 0 
       END as TOTFORMAIN
FROM Cargo_Aircraft
where AIRCRAFT_NUMBER=@regnumber;
